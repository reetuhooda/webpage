class Net(nn.Module):
    def __init__(self):
        super(Net, self).__init__()
        
        self.conv1 = nn.Conv2d(in_channels = 3, out_channels = 32, kernel_size = 3, padding = 1)
        self.conv1_bn = nn.BatchNorm2d(32)
        
        self.conv2 = nn.Conv2d(in_channels = 32, out_channels = 64, kernel_size = 3, padding = 1)
        
        self.conv3 = nn.Conv2d(in_channels = 64, out_channels = 128, kernel_size = 3, padding = 1)
        self.conv3_bn = nn.BatchNorm2d(128)
        
        self.conv4 = nn.Conv2d(in_channels = 128, out_channels = 128, kernel_size = 3, padding = 1)
        self.drop4 = nn.Dropout(p=0.05)
        
        self.conv5 = nn.Conv2d(in_channels = 128, out_channels = 256, kernel_size = 3, padding = 1)
        self.conv5_bn = nn.BatchNorm2d(256)
        
        self.conv6 = nn.Conv2d(in_channels = 256, out_channels = 256, kernel_size = 3, padding = 1)
        self.drop6 = nn.Dropout(p=0.1)
        
        self.fc1 = nn.Linear(256 * 4 * 4, 1024)
        self.fc2 = nn.Linear(1024, 512)
        self.dropfc = nn.Dropout(p=0.1)
        self.fc3 = nn.Linear(512, 10)
        
        self.pool = nn.MaxPool2d(kernel_size = 2, stride = 2)

    def forward(self, x):
        x = F.relu(self.conv1_bn(self.conv1(x)))
        x = self.pool(F.relu(self.conv2(x))) 
        
        x = F.relu(self.conv3_bn(self.conv3(x)))
        x = self.pool(F.relu(self.conv4(x)))
        
        x = self.drop4(x)
        
        x = F.relu(self.conv5_bn(self.conv5(x)))
        
        x = self.pool(F.relu(self.conv6(x)))
        
        x = self.drop6(x)
        
        x = x.view(-1, 256 * 4 * 4)  # flatten
        
        x = F.relu(self.fc1(x))
        x = F.relu(self.fc2(x))
        x = self.dropfc(x)
        x = self.fc3(x)  # the loss function already includes softmax
        return x
    
net = Net()

# To record Training and Validation Loss and Accuracy

train_loss = []
val_loss = []

train_accuracy = []
val_accuracy = []


for epoch in range(50):  # loop over the dataset multiple times

    training_loss = 0.0
    valid_loss = 0.0
    total_train = 0.0
    correct_train = 0.0
    
    for i, data in enumerate(trainloader, 0):
        
        # get the inputs; data is a list of [inputs, labels]
        inputs, labels = data[0].to(device), data[1].to(device)

        # zero the parameter gradients
        optimizer.zero_grad()

        # forward + backward + optimize
        outputs = net(inputs)
        loss = criterion(outputs, labels)
        loss.backward()
        optimizer.step()

        # print statistics
        training_loss += loss.item()
        # accuracy
        _, predicted = outputs.max(1)
        total_train += labels.size(0)
        correct_train += (predicted == labels).sum().item()
    
    # one epoch ends here
    
    # check Validation loss and accuracy
    correct_val = 0
    total_val = 0
    with torch.no_grad():
        for data in valloader:
            images, labels = data[0].to(device), data[1].to(device)
            outputs = net(images)
            loss = criterion(outputs, labels)
            valid_loss += loss.item()
            _, predicted = outputs.max(1)
            total_val += labels.size(0)
            correct_val += (predicted == labels).sum().item()
    
    # report status
    print('Iteration: %2d | Train loss: %.3f | Train Accuracy: %.3f%% | Validation loss: %.3f| Validation Accuracy: %.3f%%' %
        (epoch + 1, training_loss / len(trainloader), 100.*correct_train/total_train, valid_loss/len(valloader), 100.*correct_val/total_val))
    
    # append the loss and accuracy
    train_loss.append(training_loss / len(trainloader))
    train_accuracy.append(100.*correct_train/total_train)
    
    val_loss.append(valid_loss / len(valloader))
    val_accuracy.append(100.*correct_val/total_val)
                      

print('Finished Training')

def build_confusion_matrix(testLabel, realLabel, n):
    # plot confusion matrix
    cm=np.zeros((n,n),dtype=int)
    realLabel_np = np.array(realLabel).reshape(len(realLabel),1)
    testLabel_np = np.array(testLabel).reshape(len(testLabel),1)
    confusion_data = np.hstack((realLabel_np, testLabel_np))
    np.add.at(cm,[confusion_data[:,1:].flatten().astype(int),confusion_data[:,:1].flatten().astype(int)],1)
    CM = cm
    fig, ax = plt.subplots(figsize=(7,6))
    fig.suptitle("Confusion Matrix", fontsize=12)
    heatmap = ax.pcolor(CM, cmap=plt.cm.cool)
    ax.set_xticks(np.arange(CM.shape[1]) + 0.5, minor=False)
    ax.set_yticks(np.arange(CM.shape[0]) + 0.5, minor=False)
    ax.invert_yaxis()
    #ax.xaxis.tick_top()
    fig.colorbar(heatmap)
    ax.set_xticklabels(np.array(range(n)), minor=False)
    ax.set_yticklabels(np.array(range(n)), minor=False)
    ax.set_xlabel('Actual class')
    ax.set_ylabel('Predicted class')
    #ax.xaxis.set_label_position('top')
    for i in range(0,n):
        for j in range(0,n):
            text = ax.text(j+0.5, i+0.5, CM[i, j],
                       ha="center", va="center", color="k")
    plt.show()
